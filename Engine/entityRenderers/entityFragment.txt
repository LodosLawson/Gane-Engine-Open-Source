#version 150

// Vertex shader'dan gelen texture (kaplama) koordinatları.
in vec2 pass_textureCoords;
// Vertex shader'dan gelen parlaklık değeri (ışık hesaplamasından sonra).
in float pass_brightness;
// Vertex shader'dan gelen nesne xz düzlemi konumu.
in vec2 pass_pos;

// Ekrana basılacak son piksel rengi.
out vec4 out_colour;

// Ana kaplama (texture) resmi.
uniform sampler2D diffuseMap;
// Ekstra kaplama resmi (örneğin parlama/glow haritası).
uniform sampler2D extraMap;
// Ekstra haritanın kullanılıp kullanılmayacağını belirten değişken.
uniform float hasExtraMap;

// Ana Güneş Işığı Rengi ve Yönü
uniform vec3 lightColor;
uniform vec3 lightDirection;

// Kameranın dünya pozisyonu
uniform vec3 cameraPosition;

// Nokta/Spot Işık (Point Light) için değişkenler
in vec3 pass_worldPos;
in vec3 pass_normal;
uniform vec3 pointLightPos;
uniform vec3 pointLightColor;
uniform vec3 pointLightAttenuation; // x=sabit, y=doğrusal, z=kare formülü çarpanları

// Özel bir mesafe etkisi için merkez noktası.
const vec2 center = vec2(-2.53, 3.42);

// I want to use the glsl smoothstep function, but for some unknown reason it doesn't work on my laptop, but only when exported as a jar. Works fine in Eclipse!
// Özel bir yumuşatma geçiş fonksiyonu. GLSL'nin kendi smoothstep fonksiyonuna alternatif olarak yazılmıştır.
float smoothlyStep(float edge0, float edge1, float x){
    float t = clamp((x - edge0) / (edge1 - edge0), 0.0, 1.0);
    return t * t * (3.0 - 2.0 * t);
}

// Shader'ın ana fonksiyonu. Her bir piksel için çalışır.
void main(void){
	
	// Kaplama resminden alınan temel renk.
	vec4 diffuseColour = texture(diffuseMap, pass_textureCoords);
	
	// Eğer pikselin görünmezlik (alpha) değeri çok düşükse, pikseli çizme. (Örn: Yaprak kenarları)
	if(diffuseColour.a < 0.5){
		discard;
	}
	
	// Temel parlaklık değeri.
	float brightness = pass_brightness;
	
	// Varsayılan speküler (parlama) ayarları
	float shineDamper = 32.0;
	float reflectivity = 0.4;
	
	// Eğer ekstra pürüzlülük/parlaklık haritası varsa, değerleri her piksel için dinamik oku
	if(hasExtraMap > 0.5){
		float roughness = texture(extraMap, pass_textureCoords).r;
		// Düşük pürüzlülük (karanlık pikseller) = Pürüzsüz yüzey, yüksek parlaklık yayılımı ve yansıma gücü
		// Yüksek pürüzlülük (aydınlık pikseller) = Mat yüzey, yayvan parlaklık ve sıfıra yakın yansıma
		shineDamper = mix(128.0, 4.0, roughness);
		reflectivity = mix(0.8, 0.05, roughness);
	}
	
	// Temel renge parlaklığı ve ışık rengini uygula. (Güneş Işığı)
	vec4 sunLightColour = diffuseColour * brightness * vec4(lightColor, 1.0);
	
	// Nokta Işık / Spot Işık (Point Light) Hesaplaması
	vec3 unitNormal = normalize(pass_normal);
	vec3 toLightVector = pointLightPos - pass_worldPos;
	float distanceToLight = length(toLightVector);
	vec3 unitLightVector = normalize(toLightVector);
	
	float nDotl = max(dot(unitNormal, unitLightVector), 0.0);
	float attFactor = pointLightAttenuation.x + (pointLightAttenuation.y * distanceToLight) + (pointLightAttenuation.z * distanceToLight * distanceToLight);
	vec3 pointLightIllumination = (pointLightColor * nDotl) / attFactor;
	
	// Speküler (Dinamik Parlaklık) Yansıma Hesaplaması (Blinn-Phong)
	vec3 viewVector = normalize(cameraPosition - pass_worldPos);
	
	// 1. Güneş Işığı için Speküler Yansıma
	vec3 unitLightDirection = normalize(-lightDirection); 
	vec3 sunHalfwayVector = normalize(unitLightDirection + viewVector);
	float sunSpecularFactor = max(dot(unitNormal, sunHalfwayVector), 0.0);
	float sunSpecular = pow(sunSpecularFactor, shineDamper); 
	vec3 sunSpecularLight = sunSpecular * reflectivity * lightColor; 
 
	// 2. Nokta Işığı için Speküler Yansıma
	vec3 pointHalfwayVector = normalize(unitLightVector + viewVector);
	float pointSpecularFactor = max(dot(unitNormal, pointHalfwayVector), 0.0);
	float pointSpecular = pow(pointSpecularFactor, shineDamper);
	vec3 pointSpecularLight = (pointSpecular * reflectivity * pointLightColor) / attFactor;
	
	// Sonuç = Güneş Işığı + Nokta Işığı + Speküler Parlamalar
	vec3 totalSpecular = sunSpecularLight + pointSpecularLight;
	out_colour = sunLightColour + vec4(pointLightIllumination * diffuseColour.rgb, 0.0) + vec4(totalSpecular, 0.0);
	
	// Merkez noktadan uzaklaştıkça nesnelerin rengini beyaza (1.0) karıştırma efekti.
	float disFactor = smoothlyStep(15.0, 16.0, distance(center, pass_pos));
	out_colour.rgb = mix(out_colour.rgb, vec3(1.0), disFactor);
	
}
