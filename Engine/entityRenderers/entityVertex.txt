#version 150

// 3D modelden gelen köşe (vertex) koordinatları.
in vec3 in_position;
// 3D modelden gelen texture (kaplama) koordinatları.
in vec2 in_textureCoords;
// Yüzeyin baktığı yön (ışık hesabı için gereklidir).
in vec3 in_normal;

// Fragment shader'a aktarılacak texture koordinatları.
out vec2 pass_textureCoords;
// Fragment shader'a aktarılacak parlaklık değeri.
out float pass_brightness;
// Fragment shader'a aktarılacak dünya pozisyonu.
out vec2 pass_pos;

// Nokta ışık için gereken veriler
out vec3 pass_worldPos;
out vec3 pass_normal;

// Kamera ve nesne konumunu içeren dönüşüm matrisi.
uniform mat4 projectionViewMatrix;
// Nesnenin sadece kendisine ait konum (translation vb.) matrisi.
uniform mat4 transformationMatrix;
// Işığın yönü (güneşin geliş açısı vb.).
uniform vec3 lightDirection;
uniform float lightBrightness;
uniform float ambientLight;
// Kırpma düzlemi (su yansımaları için suyun üstünü veya altını kırpmaya yarar).
uniform vec4 plane;

// Shader'ın ana fonksiyonu. Her bir köşe (vertex) için çalışır.
void main(void){

	// Nesnenin dünya üzerindeki pozisyonu.
	vec4 worldPosition = transformationMatrix * vec4(in_position, 1.0);
	pass_pos = worldPosition.xz;
	
	// Köşenin ekrandaki son pozisyonunu hesapla.
	gl_Position = projectionViewMatrix * worldPosition;
	
	// Eğer kırpma düzleminin altındaysa/üstündeyse bu yüzeyi çizme.
	gl_ClipDistance[0] = dot(worldPosition, plane);
	
	// Kaplama koordinatlarını direkt fragment shader'a geçir.
	pass_textureCoords = in_textureCoords;
	
	// Modelin normalini dünya koordinatına çevir (Rotasyon varsa düzelt)
	vec3 worldNormal = (transformationMatrix * vec4(in_normal, 0.0)).xyz;

	// Işık yönü ve yüzey normalini kullanarak yüzeyin ne kadar ışık aldığını (parlaklığını) hesapla.
	pass_brightness = max(dot(-lightDirection, normalize(worldNormal)), 0.0) * lightBrightness + ambientLight;
	
	// Nokta ışığı (Spot/Point Light) için değerleri aktar
	pass_worldPos = worldPosition.xyz;
	// Eğer transformationMatrix'te döndürme (rotation) varsa normali de döndürmeliyiz
	pass_normal = (transformationMatrix * vec4(in_normal, 0.0)).xyz;

}
