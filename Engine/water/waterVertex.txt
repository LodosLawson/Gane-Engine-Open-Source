#version 150

in vec2 position;

out vec4 clipSpace;
out vec2 textureCoords;
out vec3 toCameraVector;
out vec2 pass_pos;
out vec3 worldPos;

uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;
uniform mat4 modelMatrix;

uniform vec3 cameraPosition;
uniform float moveFactor;
uniform float waveAmplitude;
uniform float waveFrequency;

void main(void) {
	
	vec4 baseWorldPosition = modelMatrix * vec4(position.x, 0.0, position.y, 1.0);
	
	// Gerçek 3D dalga yüksekliği (Vertex Displacement) — sinüs dalgaları
	float time = moveFactor * 100.0;
	float waveX = sin(baseWorldPosition.x * waveFrequency + time) * waveAmplitude;
	float waveZ = cos(baseWorldPosition.z * waveFrequency * 0.8 + time * 1.3) * waveAmplitude * 0.6;
	float waveHeight = waveX + waveZ;
	
	vec4 worldPosition = baseWorldPosition + vec4(0.0, waveHeight, 0.0, 0.0);
	worldPos = worldPosition.xyz;
	pass_pos = worldPosition.xz;
	
	vec4 positionRelativeToCam = viewMatrix * worldPosition;
	clipSpace = projectionMatrix * positionRelativeToCam;
	gl_Position = clipSpace;
 	
	// Dünyadaki XZ koordinatlarına göre doku haritalama (Dalga dokuları esnemez)
 	textureCoords = worldPosition.xz / 10.0;
 	toCameraVector = cameraPosition - worldPosition.xyz;
 	
}
